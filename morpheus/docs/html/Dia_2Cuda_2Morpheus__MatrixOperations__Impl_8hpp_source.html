41template <typename ExecSpace, typename Matrix, typename Vector>
42void update_diagonal(
43 Matrix& A, const Vector& diagonal,
44 typename std::enable_if_t<
45 Morpheus::is_dia_matrix_format_container_v<Matrix> &&
46 Morpheus::is_dense_vector_format_container_v<Vector> &&
47 !Morpheus::is_generic_space_v<ExecSpace> &&
48 Morpheus::is_cuda_execution_space_v<ExecSpace> &&
49 Morpheus::has_access_v<typename ExecSpace::execution_space, Matrix,
50 Vector>>* = nullptr) {
51 using index_type = typename Matrix::index_type;
52 using value_type = typename Matrix::value_type;
53
54 const size_t BLOCK_SIZE = 256;
55 const size_t MAX_BLOCKS = max_active_blocks(
56 Kernels::update_dia_diagonal_kernel<value_type, index_type>, BLOCK_SIZE,
57 (size_t)sizeof(index_type) * BLOCK_SIZE);
58 const size_t NUM_BLOCKS =
59 std::min<size_t>(MAX_BLOCKS, DIVIDE_INTO(A.nrows(), BLOCK_SIZE));
60
61 const index_type num_diagonals = A.cvalues().ncols();
62 const index_type pitch = A.cvalues().nrows();
63
64 Kernels::update_dia_diagonal_kernel<value_type, index_type>
65 <<<NUM_BLOCKS, BLOCK_SIZE, 0>>>(A.nrows(), A.ncols(), num_diagonals,
66 pitch, A.diagonal_offsets().data(),
67 A.values().data(), diagonal.data());
68#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
69 getLastCudaError("update_dia_diagonal_kernel: Kernel execution failed");
70#endif
71}
