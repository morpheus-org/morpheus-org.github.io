187 inline void activate(const formats_e index) {
188 constexpr int size = Morpheus::Impl::Variant::variant_size_v<
189 typename MatrixFormats<ValueType, Properties...>::variant>;
190 const int idx = static_cast<int>(index);
191
192 if (idx > size) {
193 std::cout << "Warning: There are " << size
194 << " available formats to switch to. "
195 << "Selecting to switch to format with index " << idx
196 << " will default to the format with index 0." << std::endl;
197 }
198 Impl::activate_impl<size, ValueType, Properties...>::activate(_formats,
199 idx);
200 }
