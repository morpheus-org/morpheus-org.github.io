48template <typename ExecSpace, typename Matrix, typename Vector1,
49 typename Vector2>
50inline void multiply(
51 const Matrix& A, const Vector1& x, Vector2& y, const bool init,
52 typename std::enable_if_t<
53 Morpheus::is_coo_matrix_format_container_v<Matrix> &&
54 Morpheus::is_dense_vector_format_container_v<Vector1> &&
55 Morpheus::is_dense_vector_format_container_v<Vector2> &&
56 !Morpheus::is_generic_space_v<ExecSpace> &&
57 Morpheus::is_openmp_execution_space_v<ExecSpace> &&
58 Morpheus::has_access_v<typename ExecSpace::execution_space, Matrix,
59 Vector1, Vector2>>* = nullptr) {
60 using value_type = typename Matrix::value_type;
61 using index_type = typename Matrix::index_type;
62
63 if (init) {
64#pragma omp parallel for
65 for (index_type n = 0; n < A.nrows(); n++) {
66 y[n] = value_type(0);
67 }
68 }
69
70#pragma omp parallel
71 {
72 const size_t num_threads = omp_get_num_threads();
73 const size_t work_per_thread = (A.nnnz() + num_threads - 1) / num_threads;
74 const size_t thread_id = omp_get_thread_num();
75 const size_t begin = work_per_thread * thread_id;
76 const size_t end = std::min(begin + work_per_thread, (size_t)A.nnnz());
77 const auto sentinel_row = A.nrows() + 1;
78
79 if (begin < end) {
80 const auto first = begin > 0 ? A.crow_indices(begin - 1) : sentinel_row;
81 const auto last = end < A.nnnz() ? A.crow_indices(end) : sentinel_row;
82 auto n = begin;
83
84 // handle non-overlapping rows
85 for (; n < end && A.crow_indices(n) != last; n++) {
86 y[A.crow_indices(n)] += A.cvalues(n) * x[A.ccolumn_indices(n)];
87 }
88
89 // handle row overlap with previous thread
90 if (first != sentinel_row) {
91 value_type partial_sum = 0;
92 for (; n < end && A.crow_indices(n) == first; n++) {
93 partial_sum += A.cvalues(n) * x[A.ccolumn_indices(n)];
94 }
95#pragma omp atomic
96 y[first] += partial_sum;
97 }
98
99 // handle row overlap with following thread
100 if (last != sentinel_row) {
101 value_type partial_sum = 0;
102 for (; n < end; n++) {
103 partial_sum += A.cvalues(n) * x[A.ccolumn_indices(n)];
104 }
105#pragma omp atomic
106 y[last] += partial_sum;
107 }
108 }
109 }
110}
