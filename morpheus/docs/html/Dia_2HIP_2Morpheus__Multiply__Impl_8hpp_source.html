40template <typename ExecSpace, typename Matrix, typename Vector>
41inline void multiply(
42 const Matrix& A, const Vector& x, Vector& y, const bool init,
43 typename std::enable_if_t<
44 Morpheus::is_dia_matrix_format_container_v<Matrix> &&
45 Morpheus::is_dense_vector_format_container_v<Vector> &&
46 !Morpheus::is_generic_space_v<ExecSpace> &&
47 Morpheus::is_hip_execution_space_v<ExecSpace> &&
48 Morpheus::has_access_v<typename ExecSpace::execution_space, Matrix,
49 Vector>>* = nullptr) {
50 using index_type = typename Matrix::index_type;
51 using value_type = typename Matrix::value_type;
52
53 const size_t BLOCK_SIZE = 256;
54 const size_t MAX_BLOCKS = max_active_blocks(
55 Kernels::spmv_dia_kernel<index_type, value_type, BLOCK_SIZE>, BLOCK_SIZE,
56 (size_t)sizeof(index_type) * BLOCK_SIZE);
57 const size_t NUM_BLOCKS =
58 std::min<size_t>(MAX_BLOCKS, DIVIDE_INTO(A.nrows(), BLOCK_SIZE));
59
60 const index_type* D = A.cdiagonal_offsets().data();
61 const value_type* V = A.cvalues().data();
62 const value_type* x_ptr = x.data();
63 value_type* y_ptr = y.data();
64
65 const index_type num_diagonals = A.cvalues().ncols();
66 const index_type pitch = A.cvalues().nrows();
67
68 if (num_diagonals == 0) {
69 // empty matrix
70 return;
71 }
72
73 if (init) {
74 y.assign(y.size(), 0);
75 }
76
77 Kernels::spmv_dia_kernel<index_type, value_type, BLOCK_SIZE>
78 <<<NUM_BLOCKS, BLOCK_SIZE, 0>>>(A.nrows(), A.ncols(), num_diagonals,
79 pitch, D, V, x_ptr, y_ptr);
80
81#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
82 getLastHIPError("spmv_dia_kernel: Kernel execution failed");
83#endif
84}
