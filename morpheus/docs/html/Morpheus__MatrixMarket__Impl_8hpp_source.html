49 inline void tokenize(std::vector<std::string>& tokens, const std::string& str,
50  const std::string& delimiters = "\n\r\t ") {
51  // Skip delimiters at beginning.
52  std::string::size_type lastPos = str.find_first_not_of(delimiters, 0);
53  // Find first "non-delimiter".
54  std::string::size_type pos = str.find_first_of(delimiters, lastPos);
55 
56  while (std::string::npos != pos || std::string::npos != lastPos) {
57  // Found a token, add it to the vector.
58  tokens.push_back(str.substr(lastPos, pos - lastPos));
59  // Skip delimiters. Note the "not_of"
60  lastPos = str.find_first_not_of(delimiters, pos);
61  // Find next "non-delimiter"
62  pos = str.find_first_of(delimiters, lastPos);
63  }
64 }

95  banner.type != "integer" && banner.type != "pattern")
96  throw Morpheus::IOException("invalid MatrixMarket data type [" +
97  banner.type + "]");
98

100  banner.symmetry != "hermitian" && banner.symmetry != "skew-symmetric")
101  throw Morpheus::IOException("invalid MatrixMarket symmetry [" +
102  banner.symmetry + "]");

105 template <typename ValueType, typename... Properties, typename Stream>
106 void read_coordinate_stream(Morpheus::CooMatrix<ValueType, Properties...>& coo,
107  Stream& input, const matrix_market_banner& banner) {
108  using IndexType =
109  typename Morpheus::CooMatrix<ValueType, Properties...>::index_type;
110  // read file contents line by line
111  std::string line;
112 
113  // skip over banner and comments
114  do {
115  std::getline(input, line);
116  } while (line[0] == '%');
117 
118  // line contains [num_rows num_columns num_entries]
119  std::vector<std::string> tokens;
120  Impl::tokenize(tokens, line);
121 
122  if (tokens.size() != 3)
123  throw Morpheus::IOException("invalid MatrixMarket coordinate format");
124 
125  IndexType num_rows, num_cols, num_entries;
126 
127  std::istringstream(tokens[0]) >> num_rows;
128  std::istringstream(tokens[1]) >> num_cols;
129  std::istringstream(tokens[2]) >> num_entries;
130 
131  coo.resize(num_rows, num_cols, num_entries);
132 
133  IndexType num_entries_read = 0;
134 
135  // read file contents
136  if (banner.type == "pattern") {
137  while (num_entries_read < coo.nnnz() && !input.eof()) {
138  input >> coo.row_indices(num_entries_read);
139  input >> coo.column_indices(num_entries_read);
140  num_entries_read++;
141  }
142 
143  auto values_begin = coo.values().data();
144  auto values_end = coo.values().data() + coo.values().size();
145  std::fill(values_begin, values_end, ValueType(1));
146  } else if (banner.type == "real" || banner.type == "integer") {
147  while (num_entries_read < coo.nnnz() && !input.eof()) {
148  ValueType real;
149 
150  input >> coo.row_indices(num_entries_read);
151  input >> coo.column_indices(num_entries_read);
152  input >> real;
153 
154  coo.values(num_entries_read) = real;
155  num_entries_read++;
156  }
157  } else if (banner.type == "complex") {
158  throw Morpheus::NotImplementedException(
159  "Morpheus does not currently support complex "
160  "matrices");
161 
162  } else {
163  throw Morpheus::IOException("invalid MatrixMarket data type");
164  }
165 
166  if (num_entries_read != coo.nnnz())
167  throw Morpheus::IOException(
168  "unexpected EOF while reading MatrixMarket entries");
169 
170  // check validity of row and column index data
171  if (coo.nnnz() > 0) {
172  auto row_indices_begin = coo.row_indices().data();
173  auto row_indices_end = coo.row_indices().data() + coo.row_indices().size();
174  IndexType min_row_index =
175  *std::min_element(row_indices_begin, row_indices_end);
176  IndexType max_row_index =
177  *std::max_element(row_indices_begin, row_indices_end);
178 
179  auto column_indices_begin = coo.column_indices().data();
180  auto column_indices_end =
181  coo.column_indices().data() + coo.column_indices().size();
182  IndexType min_col_index =
183  *std::min_element(column_indices_begin, column_indices_end);
184  IndexType max_col_index =
185  *std::max_element(column_indices_begin, column_indices_end);
186 
187  if (min_row_index < 1)
188  throw Morpheus::IOException("found invalid row index (index < 1)");
189  if (min_col_index < 1)
190  throw Morpheus::IOException("found invalid column index (index < 1)");
191  if (max_row_index > coo.nrows())
192  throw Morpheus::IOException("found invalid row index (index > num_rows)");
193  if (max_col_index > coo.ncols())
194  throw Morpheus::IOException(
195  "found invalid column index (index > num_columns)");
196  }
197 
198  // convert base-1 indices to base-0
199  for (IndexType n = 0; n < coo.nnnz(); n++) {
200  coo.row_indices(n) -= 1;
201  coo.column_indices(n) -= 1;
202  }
203 
204  // expand symmetric formats to "general" format
205  if (banner.symmetry != "general") {
206  IndexType off_diagonals = 0;
207 
208  for (IndexType n = 0; n < coo.nnnz(); n++)
209  if (coo.row_indices(n) != coo.column_indices(n)) off_diagonals++;
210 
211  IndexType general_num_entries = coo.nnnz() + off_diagonals;
212 
213  Morpheus::CooMatrix<ValueType, Properties...> general(num_rows, num_cols,
214  general_num_entries);
215 
216  if (banner.symmetry == "symmetric") {
217  IndexType nnz = 0;
218 
219  for (IndexType n = 0; n < coo.nnnz(); n++) {
220  // copy entry over
221  general.row_indices(nnz) = coo.row_indices(n);
222  general.column_indices(nnz) = coo.column_indices(n);
223  general.values(nnz) = coo.values(n);
224  nnz++;
225 
226  // duplicate off-diagonals
227  if (coo.row_indices(n) != coo.column_indices(n)) {
228  general.row_indices(nnz) = coo.column_indices(n);
229  general.column_indices(nnz) = coo.row_indices(n);
230  general.values(nnz) = coo.values(n);
231  nnz++;
232  }
233  }
234  } else if (banner.symmetry == "hermitian") {
235  throw Morpheus::NotImplementedException(
236  "MatrixMarket I/O does not currently support hermitian matrices");
237  // TODO
238  } else if (banner.symmetry == "skew-symmetric") {
239  // TODO
240  throw Morpheus::NotImplementedException(
241  "MatrixMarket I/O does not currently support skew-symmetric "
242  "matrices");
243  }
244 
245  // store full matrix in coo
246  coo = general;
247  } // if (banner.symmetry != "general")
248  // sort indices by (row,column)
249  Morpheus::sort_by_row_and_column(coo);
250 }

253 void read_array_stream(Morpheus::DenseMatrix<ValueType, Properties...>& mtx,
254  Stream& input, const matrix_market_banner& banner) {
255  using IndexType =
256  typename Morpheus::DenseMatrix<ValueType, Properties...>::index_type;
257  // read file contents line by line
258  std::string line;
259 
260  // skip over banner and comments
261  do {
262  std::getline(input, line);
263  } while (line[0] == '%');
264 
265  std::vector<std::string> tokens;
266  Impl::tokenize(tokens, line);
267 
268  if (tokens.size() != 2)
269  throw Morpheus::IOException("invalid MatrixMarket array format");
270 
271  IndexType num_rows, num_cols;
272 
273  std::istringstream(tokens[0]) >> num_rows;
274  std::istringstream(tokens[1]) >> num_cols;
275 
276  Morpheus::DenseMatrix<ValueType, Properties...> dense(num_rows, num_cols);
277 
278  size_t num_entries = num_rows * num_cols;
279  size_t num_entries_read = 0;
280  IndexType nrows = 0, ncols = 0;
281 
282  // read file contents
283  if (banner.type == "pattern") {
284  throw Morpheus::NotImplementedException(
285  "pattern array MatrixMarket format is not supported");
286  } else if (banner.type == "real" || banner.type == "integer") {
287  while (num_entries_read < num_entries && !input.eof()) {
288  double real;
289 
290  input >> real;
291 
292  dense(nrows, ncols) = ValueType(real);
293 
294  num_entries_read++;
295  ncols++;
296  if (num_entries_read % ncols == 0) {
297  nrows++;
298  ncols = 0;
299  }
300  }
301  } else if (banner.type == "complex") {
302  throw Morpheus::NotImplementedException("complex type is not supported");
303  } else {
304  throw Morpheus::IOException("invalid MatrixMarket data type");
305  }
306 
307  if (num_entries_read != num_entries)
308  throw Morpheus::IOException(
309  "unexpected EOF while reading MatrixMarket entries");
310 
311  if (banner.symmetry != "general")
312  throw Morpheus::NotImplementedException(
313  "only general array symmetric MatrixMarket format is supported");
314 
315  mtx = dense;
316  // Morpheus::copy(dense, mtx);
317 }
