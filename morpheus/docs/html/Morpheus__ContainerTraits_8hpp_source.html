183 using IndexType = typename std::conditional_t<

185 typename prop::index_type, int>;

187 using ExecutionSpace = typename std::conditional_t<
188 !std::is_same_v<typename prop::execution_space, void>,
189 typename prop::execution_space, Kokkos::DefaultExecutionSpace>;
190
191 using MemorySpace = typename std::conditional_t<
192 !std::is_same_v<typename prop::memory_space, void>,
193 typename prop::memory_space, typename ExecutionSpace::memory_space>;
194
195 using ArrayLayout = typename std::conditional_t<
196 !std::is_same_v<typename prop::array_layout, void>,
197 typename prop::array_layout, typename ExecutionSpace::array_layout>;
198
199 using HostMirrorSpace = typename std::conditional<
200 !std::is_same<typename prop::HostMirrorSpace, void>::value,
201 typename prop::HostMirrorSpace,
202 typename Kokkos::Impl::HostMirror<ExecutionSpace>::Space>::type;
203
204 using MemoryTraits = typename std::conditional<
205 !std::is_same<typename prop::memory_traits, void>::value,
206 typename prop::memory_traits, typename Kokkos::MemoryManaged>::type;

208 // Check the validity of ValueType

210 "ValueType must be an arithmetic type such as int or double");
