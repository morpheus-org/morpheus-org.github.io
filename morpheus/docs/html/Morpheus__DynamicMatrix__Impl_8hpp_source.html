177template <size_t I, class ValueType, typename... Properties>
178struct activate_impl {
179 using variant = typename MatrixFormats<ValueType, Properties...>::variant;
180 using type_list = typename MatrixFormats<ValueType, Properties...>::type_list;
181
182 static void activate(variant &A, size_t idx) {
183 if (idx == I - 1) {
184 A = typename type_list::template type<I - 1>{};
185 } else {
186 activate_impl<I - 1, ValueType, Properties...>::activate(A, idx);
187 }
188 }
189};
