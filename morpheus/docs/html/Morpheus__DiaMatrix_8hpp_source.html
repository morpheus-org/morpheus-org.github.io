388 inline void resize(const index_type num_rows, const index_type num_cols,
389 const index_type num_entries,
390 const index_type num_diagonals,
391 const index_type alignment = 32) {
392 base::resize(num_rows, num_cols, num_entries);
393 _ndiags = num_diagonals;
394 _alignment = alignment;
395
396 if (Impl::exceeds_tolerance(num_rows, num_entries, _ndiags)) {
397 throw Morpheus::FormatConversionException(
398 "DiaMatrix fill-in would exceed maximum tolerance");
399 }
400
401 _diagonal_offsets.resize(num_diagonals);
402 _values.resize(Impl::get_pad_size<index_type>(num_rows, alignment),
403 num_diagonals);
404 }
