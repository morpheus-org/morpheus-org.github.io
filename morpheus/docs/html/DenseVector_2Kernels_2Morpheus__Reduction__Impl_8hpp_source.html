46 // Utility class used to avoid linker errors with extern
47 // unsized shared memory arrays with templated type
48 template <class T>
49 struct SharedMemory {
50  __device__ inline operator T *() {
51  extern __shared__ int __smem[];
52  return (T *)__smem;
53  }
54 
55  __device__ inline operator const T *() const {
56  extern __shared__ int __smem[];
57  return (T *)__smem;
58  }
59 };

76 template <class T>
77 __device__ __forceinline__ T warpReduceSum(unsigned int mask, T mySum) {
78  for (int offset = warpSize / 2; offset > 0; offset /= 2) {
79  mySum += __shfl_down_sync(mask, mySum, offset);
80  }
81  return mySum;
82 }

88 __device__ __forceinline__ int warpReduceSum<int>(unsigned int mask,

95 /*
96  Parallel sum reduction using shared memory
97  - takes log(n) steps for n input elements
98  - uses n threads
99  - only works for power-of-2 arrays
100 */
101 
102 template <typename ValueType, unsigned int blockSize, bool nIsPow2>
103 __global__ void reduce_kernel(const ValueType *__restrict__ g_idata,
104  ValueType *__restrict__ g_odata, unsigned int n) {
105  ValueType *sdata = SharedMemory<ValueType>();
106 
107  // perform first level of reduction,
108  // reading from global memory, writing to shared memory
109  unsigned int tid = threadIdx.x;
110  unsigned int gridSize = blockSize * gridDim.x;
111  unsigned int maskLength = (blockSize & 31); // 31 = warpSize-1
112  maskLength = (maskLength > 0) ? (32 - maskLength) : maskLength;
113  const unsigned int mask = (0xffffffff) >> maskLength;
114 
115  ValueType mySum = 0;
116 
117  // we reduce multiple elements per thread. The number is determined by the
118  // number of active thread blocks (via gridDim). More blocks will result
119  // in a larger gridSize and therefore fewer elements per thread
120  if (nIsPow2) {
121  unsigned int i = blockIdx.x * blockSize * 2 + threadIdx.x;
122  gridSize = gridSize << 1;
123 
124  while (i < n) {
125  mySum += g_idata[i];
126  // ensure we don't read out of bounds -- this is optimized away for
127  // powerOf2 sized arrays
128  if ((i + blockSize) < n) {
129  mySum += g_idata[i + blockSize];
130  }
131  i += gridSize;
132  }
133  } else {
134  unsigned int i = blockIdx.x * blockSize + threadIdx.x;
135  while (i < n) {
136  mySum += g_idata[i];
137  i += gridSize;
138  }
139  }
140 
141  // Reduce within warp using shuffle or reduce_add if T==int & CUDA_ARCH ==
142  // SM 8.0
143  mySum = warpReduceSum<ValueType>(mask, mySum);
144 
145  // each thread puts its local sum into shared memory
146  if ((tid % warpSize) == 0) {
147  sdata[tid / warpSize] = mySum;
148  }
149 
150  __syncthreads();
151 
152  const unsigned int shmem_extent =
153  (blockSize / warpSize) > 0 ? (blockSize / warpSize) : 1;
154  const unsigned int ballot_result = __ballot_sync(mask, tid < shmem_extent);
155  if (tid < shmem_extent) {
156  mySum = sdata[tid];
157  // Reduce final warp using shuffle or reduce_add if T==int & CUDA_ARCH ==
158  // SM 8.0
159  mySum = warpReduceSum<ValueType>(ballot_result, mySum);
160  }
161 
162  // write result for this block to global mem
163  if (tid == 0) {
164  g_odata[blockIdx.x] = mySum;
165  }
166 }
