87 // Compute the number of threads and blocks to use for the given reduction
88 // kernel. We set threads / block to the minimum of maxThreads and n/2.
89 // We observe the maximum specified number of blocks, because
90 // each thread in that kernel can process a variable number of elements.
91 template <typename IndexType>
92 void getNumBlocksAndThreads(
93  IndexType n, IndexType maxBlocks, IndexType maxThreads, IndexType &blocks,
94  IndexType &threads,
95  typename std::enable_if<std::is_integral<IndexType>::value>::type * =
96  nullptr) {
97  // get device capability, to avoid block/grid size exceed the upper bound
98  cudaDeviceProp prop;
99  int device;
100  checkCudaErrors(cudaGetDevice(&device));
101  checkCudaErrors(cudaGetDeviceProperties(&prop, device));
102 
103  threads = (n < maxThreads * 2) ? nextPow2((n + 1) / 2) : maxThreads;
104  blocks = (n + (threads * 2 - 1)) / (threads * 2);
105 
106  if ((float)threads * blocks >
107  (float)prop.maxGridSize[0] * prop.maxThreadsPerBlock) {
108  printf("n is too large, please choose a smaller number!\n");
109  }
110 
111  if (blocks > prop.maxGridSize[0]) {
112  printf(
113  "Grid size <%d> exceeds the device capability <%d>, set block size as "
114  "%d (original %d)\n",
115  blocks, prop.maxGridSize[0], threads * 2, threads);
116 
117  blocks /= 2;
118  threads *= 2;
119  }
120 
121  blocks = min(maxBlocks, blocks);
122 }
