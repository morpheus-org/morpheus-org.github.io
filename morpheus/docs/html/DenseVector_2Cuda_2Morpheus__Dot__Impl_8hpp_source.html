47template <typename ExecSpace, typename Vector1, typename Vector2>
48typename Vector1::value_type dot(
49 const typename Vector1::index_type n, const Vector1& x, const Vector2& y,
50 typename std::enable_if_t<
51 Morpheus::is_dense_vector_format_container_v<Vector1> &&
52 Morpheus::is_dense_vector_format_container_v<Vector2> &&
53 !Morpheus::is_generic_space_v<ExecSpace> &&
54 Morpheus::is_cuda_execution_space_v<ExecSpace> &&
55 Morpheus::has_access_v<typename ExecSpace::execution_space, Vector1,
56 Vector2>>* = nullptr) {
57 using index_type = typename Vector1::index_type;
58 using value_type = typename Vector1::non_const_value_type;
59
60 value_type local_result;
61
62#ifdef MORPHEUS_ENABLE_TPL_CUBLAS
63 cublasdotspace.init();
64 cublasdotspace.allocate<value_type>(1);
65 index_type incx = 1, incy = 1;
66 cublasDdot(cublasdotspace.handle(), n, x.data(), incx, y.data(), incy,
67 cublasdotspace.data<value_type>());
68
69 cudaMemcpy(&local_result, cublasdotspace.data<value_type>(),
70 sizeof(value_type), cudaMemcpyDeviceToHost);
71#else
72 cudotspace.allocate<value_type>(n);
73
74 Kernels::dot_kernel_part1<256, value_type, index_type>
75 <<<256, 256>>>(n, x.data(), y.data(), cudotspace.data<value_type>());
76#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
77 getLastCudaError("dot: Kernel execution failed");
78#endif
79
80 Kernels::dot_kernel_part2<256, value_type>
81 <<<1, 256>>>(cudotspace.data<value_type>());
82#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
83 getLastCudaError("dot: Kernel execution failed");
84#endif
85
86 cudaMemcpy(&local_result, cudotspace.data<value_type>(), sizeof(value_type),
87 cudaMemcpyDeviceToHost);
88#endif // MORPHEUS_ENABLE_TPL_CUBLAS
89
90 return local_result;
91}
